374inline bool Vertex::removeEdge(const Web& in) {
375 bool removedEdge = false;
376 auto it = adj.begin();
377 while (it != adj.end()) {
378 Edge *edge = *it;
379 Vertex *dest = edge->getDest();
380 if (dest->getInfo() == in) {
381 it = adj.erase(it);
382 deleteEdge(edge);
383 removedEdge = true;
384 }
385 else {
386 ++it;
387 }
388 }
389 return removedEdge;
390}

547inline void Graph::createEdges() {
548 // Create a vector of all webs, sorted by starting line number
549 std::sort(vertexSet.begin(), vertexSet.end(), [] (const Vertex* v1, const Vertex* v2) {;
550 return v1->getInfo().getFirstLineNum() < v2->getInfo().getFirstLineNum();
551 });
552
553 std::set<Web> activeWebs;
554
555 for (const Vertex* vertex : vertexSet) {
556 Web curWeb = vertex->getInfo();
557
558 auto iter = activeWebs.begin();
559 while (iter != activeWebs.end()) {
560 if (curWeb.getFirstLineNum() >= iter->getLastLineNum()) {
561 iter = activeWebs.erase(iter);
562 } else {
563 if (this->addBidirectionalEdge(curWeb, *iter)) ++iter;
564 }
565 }
566 activeWebs.insert(curWeb);
567 }
568}

600inline bool Graph::removeVertex(const Web &in) {
601 for (auto it = vertexSet.begin(); it != vertexSet.end(); it++) {
602 if ((*it)->getInfo() == in) {
603 auto v = *it;
604 v->removeOutgoingEdges();
605 for (auto u : vertexSet) {
606 u->removeEdge(v->getInfo());
607 }
608 vertexSet.erase(it);
609 delete v;
610 return true;
611 }
612 }
613 return false;

635inline bool Graph::addBidirectionalEdge(const Web &source, const Web &dest) const {
636 auto v1 = findVertex(source);
637 auto v2 = findVertex(dest);
638 if (v1 == nullptr || v2 == nullptr)
639 return false;
640 auto e1 = v1->addEdge(v2);
641 auto e2 = v2->addEdge(v1);
642 e1->setReverse(e2);
643 e2->setReverse(e1);
644 return true;
645}
